Function WriteToFile (FileUrlDst, Str, CharSet)
Set stm = CreateObject("Adodb.Stream")
stm.Type = 2
stm.mode = 3
stm.charset = CharSet
stm.Open
stm.WriteText Str
stm.SaveToFile FileUrlDst, 2
stm.flush
stm.Close
Set stm = Nothing
End Function
